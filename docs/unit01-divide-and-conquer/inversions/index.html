int combine(int A[], int l, int mid, int r) {  int *tmp = new int[r - l + 1];  int i = l, j = mid, k = 0, count_total = 0;  while (i < mid || j <= r) {    if (j > r || (i < mid && A[i] <= A[j])) {      tmp[k++] = A[i++];      count_total += j - mid;    } else {      tmp[k++] = A[j++];    }  }  for (int i = l; i <= r; i++) A[i] = tmp[i - l];  delete[] tmp;  return count_total;}
int count_inversions_impl(int A[], int l, int r) {  if (l >= r) return 0;  int mid = (l + r)/2;  int count_left = count_inversions_impl(A, l, mid);  int count_right = count_inversions_impl(A, mid+1, r);  int count_mid = combine(A, l, mid+1, r);  return count_left + count_right + count_mid;}
int count_inversions(int A[], int n) {  return count_inversions_impl(A, 0, n - 1);}
